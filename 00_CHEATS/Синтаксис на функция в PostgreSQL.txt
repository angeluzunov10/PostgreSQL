CREATE OR REPLACE FUNCTION fn_името_на_функцията(типовете данни, които ще приема)
RETURNS типа данни, който ще върне функцията AS
$$	
	DECLARE
		името на променливата и от какъв тип е;
		.........за всяка променлива си пишем име;
	BEGIN	
		тялото на функцията;
		RETURN какво искаме да ни върне;
	END;
$$
LANGUAGE plpgsql





може избора в тялото да го пъхнем в променлива, декларирана в DECLARE, като напишем INTO името на променливата
